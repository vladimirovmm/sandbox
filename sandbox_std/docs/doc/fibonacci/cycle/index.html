Подсчёта числа Фибоначчи через цикл for + swap

pub fn fast_cycle_for(n: u64) -> u64 {
    if n <= 1 {
        return 1;
    }

    let mut a = 0;
    let mut b = 1;

    for _ in 0..n {
        a += b;
        std::mem::swap(&mut a, &mut b);
    }

    b
}
